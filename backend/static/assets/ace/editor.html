#include<iostream>
#include<vector>
#include<algorithm>
#include<cstdio>
using namespace std;
int main(){
    int n;cin>>n;
    int x;
    vector<int> input;
    for(int i=0;i<n;i++){
        cin>>x;
        input.push_back(x);
    }
    sort(input.begin(),input.end());
    for(int i=0;i<input.size();i++)
        cout<<input[i]<<" ";
    cout<<endl;
    return 0;
}
